int lseek(struct file *f, int offset, int whence)
{
  struct inode *inode = f->ip;
  switch (whence)
  {
  case SEEK_SET:
    f->off = offset;
    break;
  case SEEK_CUR:
    if (offset != 0)
      f->off += offset;
    break;
  case SEEK_END:
    f->off = inode->size + offset;
    break;
  }
  offset = f->off;
  return offset;
}
